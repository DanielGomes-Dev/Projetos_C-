Namespace;
    Não podemos ter duas classes com mesmo nome em um namespace;
    O ideal é ter apenas um namespace e uma classe por arquivo;
    Durante a execução, todos os arquivos do C# serão unificados;

    Sobrará apenas a divisão lógica, que são os namespace
    usar using para chamar namespace em outro arquivo

Structs
    struct

    struct Product 
    {
        //Propriedades;
        //Metodos;
    }

    struct Product
    {
        //Propriedade;
        public int Id;
        public float Price;

        //Metodo Construtor;
        public Product(int id, float price){
            Id = id;
            Price = price
        }

        //Metodos;
        public float PriceInDolar(Float Dolar){
            return Price * dolar;
        }

    }

    var product = new Product(1, 10.10f);
    product.Id = 1;
    product.Price = 10.11f;

    Metodo Construtor é um metodo com o mesmo nome da estrutura;
